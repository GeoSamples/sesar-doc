CREATE OR REPLACE FUNCTION sesar_fill_igsn_count() 
  RETURNS boolean AS 
$$ 
DECLARE
 curInfo RECORD;  
BEGIN
    -- Loop through each user_code.
    FOR curInfo IN
        select user_code 
        from sesar_user_code
    LOOP             
        perform sesar_update_igsn_count(curInfo.user_code);
    End LOOP;
    RETURN true::boolean;
END;
$$
  LANGUAGE plpgsql VOLATILE 
  COST 100;
